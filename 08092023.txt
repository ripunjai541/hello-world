2.Amicable number


n1 = int(input())
n2=int(input())
l1=[]
l2=[]
for i in range(1,(n1//2)+1):
  if n1%i==0:
    l1.append(i)
for i in range(1,(n2//2)+1):
  if n2%i==0:
    l2.append(i)
if sum(l1)==n2 and sum(l2)==n1:
  print("amacable number")
else:
  print("not amacable number")
